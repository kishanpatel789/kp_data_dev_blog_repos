Config Files

What's a Config File?

JSON


  {
    "app": {
      "name": "DefenseAgainstTheDarkArts",
      "version": "0.0.1",
      "debug": false,
      "secret_key": "ssshhh-my-api-key-alohomora"
    },
    "server": {
      "host": "0.0.0.0",
      "port": 8080
    },
    "database": {
      "engine": "postgresql",
      "host": "db.hogwarts.edu",
      "port": 5432,
      "username": "dobby",
      "password": "sssshhh-secret-database-password",
      "name": "public",
      "pool_size": 10
    },
    "logging": {
      "level": "info",
      "format": "json",
      "output": "logs/app.log"
    }
  }

JSON


{
  "app": {
    "name": "DefenseAgainstTheDarkArts",
    "version": "0.0.1",
    "environment": "dev",
    "debug": false,
    "secret_key": "ssshhh-my-api-key-alohomora"
  },
  "server": {
    "host": "0.0.0.0",
    "port": 8080
  }
}

YAML


app:
  name: DefenseAgainstTheDarkArts
  version: 0.0.1
  environment: dev
  debug: true
  secret_key: ssshhh-my-api-key-alohomora
server:
  host: 0.0.0.0
  port: 8080

YAML


app:
  name: DefenseAgainstTheDarkArts
  version: 0.0.1
  environment: dev  # options: dev, staging, production
  debug: true       # WARNING: set this to false before deploying
  secret_key: ssshhh-my-api-key-alohomora

# network configuration
server:
  host: 0.0.0.0
  port: 8080
