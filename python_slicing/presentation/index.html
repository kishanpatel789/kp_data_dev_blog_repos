Python Slicing

Behind the Scenes

When using the [] operator...

  1. Python generates a slice object.
  2. That slice object is sent to the sequence's __getitem__ method.

Re-Use Code

JSON


{
  "services": {
    "api": {
      "host": "0.0.0.0",
      "port": 8080,
      "logging": {
        "level": "info",
        "format": "json",
        "output": "logs/app.log"
      }
    },
    "worker": {
      "concurrency": 4,
      "logging": {
        "level": "info",
        "format": "json",
        "output": "logs/app.log"
      }
    }
  }
}

YAML


services:
  api:
    host: 0.0.0.0
    port: 8080
    logging: 
      level: info
      format: json
      output: logs/app.log
  worker:
    concurrency: 4
    logging: 
      level: info
      format: json
      output: logs/app.log

YAML


services:
  api:
    host: 0.0.0.0
    port: 8080
    logging: &default_logging # anchor logging node
      level: info
      format: json
      output: logs/app.log
  worker:
    concurrency: 4
    logging: *default_logging  # reuse logging settings
