    if user.is_admin:
        print(f"Found one admin: {user}")

Boolean Flag Approach

admin_found = False # flag to store success or failure

for user in users:
    if user.is_admin:
        print(f"Found one admin: {user}")
        admin_found = True  # update flag 
        break

if not admin_found:
    print("No admin user found!")
